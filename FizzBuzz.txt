import java.util.Scanner;

public class FizzBuzz {
	public static void main(String[]args) {
		Scanner scan= new Scanner(System.in);
		String name =scan.nextLine();
		String name1 =scan.nextLine();
		String name2 =scan.nextLine();
		
		System.out.println(fizzString(name));
		System.out.println(fizzString(name1));
		System.out.println(fizzString(name2));
	
	}
	public static String fizzString(String str) {
		  if (str.substring(0,1).equals("f")&&str.substring(str.length()-1).equals("b")){
		    return "FizzBuzz";
		    
		  }
		  if(str.substring(0,1).equals("f")){
		    return "Fizz";
		  }else if(str.substring(str.length()-1).equals("b")){
		    return "Buzz";
		  }
		  return str;
		}
	


}
